@import org.knora.webapi.IRI

@**
 * Gets the administrative permission information attached to a group, given the project and group IRI.
 *
 * @param triplestore the name of the triplestore being used.
 * @param projectIri the project's IRI.
 * @param groupIri the group's IRI.
 *@
@(triplestore: String,
  projectIri: IRI,
  groupIri: IRI)

prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
prefix knora-base: <http://www.knora.org/ontology/knora-base#>

SELECT ?s ?p ?o

@* Ensure that inference is not used in this query. *@
@if(triplestore.startsWith("graphdb")) {
    FROM <http://www.ontotext.com/explicit>
}

WHERE {
    BIND(IRI("@projectIri") as ?pIri)
    BIND(IRI("@groupIri") as ?gIri)

    ?s rdf:type knora-base:AdministrativePermission .
    ?s knora-base:forProject ?pIri .
    ?s knora-base:forGroup ?gIri .
    ?s ?p ?o .
}
